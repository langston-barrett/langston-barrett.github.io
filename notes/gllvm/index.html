GLLVM

A tool for compiling C/C++ to LLVM bitcode. See also WLLVM.

.bca Files

Just pass the -b flag to get-bc.

Compiling Linux

GLLVM can compile the Linux kernel.

# docker build -t gllvm-linux . 
# docker run --rm --workdir=/work --mount type=bind,src=$PWD,target=/work -it gllvm-linux
FROM ubuntu:20.04

# Configuration
# Newer versions of Linux require "asm-goto", which isn't in LLVM 8...
ENV KERNEL_VERSION=4.14.39
# ENV KERNEL_VERSION=4.19.182
ENV LLVM_VERSION=7

# Static environment
ENV LLVM_COMPILER=clang
ENV DEBIAN_FRONTEND=noninteractive

# Get the kernel
RUN apt-get update -qq -y
RUN apt-get install -qq -y curl build-essential
RUN curl -O "https://cdn.kernel.org/pub/linux/kernel/v4.x/linux-${KERNEL_VERSION}.tar.xz"
RUN tar xf linux-${KERNEL_VERSION}.tar.xz
WORKDIR linux-${KERNEL_VERSION}

# Install build dependencies
RUN apt-get install -qq -y dbus-x11 git golang-go flex bison bc libncurses5-dev libelf-dev libssl-dev
RUN apt-get install -qq -y llvm-${LLVM_VERSION} clang-${LLVM_VERSION} libclang-${LLVM_VERSION}-dev lld-${LLVM_VERSION}

# Build it
ENV GOPATH=/gopath
ENV PATH=$PATH:$GOPATH/bin:/usr/lib/llvm-${LLVM_VERSION}/bin
RUN go get github.com/SRI-CSL/gllvm/cmd/...
COPY tinyconfig .config
RUN make -j CC=gclang HOSTCC=gclang LLVM=1 CONFIG_CC_IS_CLANG=1
RUN get-bc -m -b built-in.o && get-bc -m vmlinux
CMD cp /linux-${KERNEL_VERSION}/*.bc /work
